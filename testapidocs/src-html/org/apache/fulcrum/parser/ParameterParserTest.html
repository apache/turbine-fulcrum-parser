164    @Test
165    public void testRequests() throws Exception
166    {
167        
168        assertTrue(parameterParser.getRequest() == null);
169        HttpServletRequest req = getMockRequest();
170        
171        parameterParser.setRequest(req);
172        
173        parameterParser.add("xy", 35);
174        
175        assertTrue(parameterParser.getRequest().getServerPort() == 8080);
176        assertTrue(parameterParser.getString("xy").equals("35"));
177        
178        parserService.putParser(parameterParser);
179        assertTrue(parameterParser.getRequest() == null);
180        assertTrue(parameterParser.getString("xy") == null);
181
182        
183        ParameterParser parameterParser2 = parserService.getParser(DefaultParameterParser.class);
184        assertTrue(parameterParser2.getRequest() == null);
185        parameterParser2.add("ab", 7);
186        
187        // only one object in pool, which is refetched
188        assertTrue(parameterParser.getString("ab").equals("7"));
189        assertTrue(parameterParser.getString("xy") == null);
190        
191        assertTrue(parameterParser2.equals(parameterParser),parameterParser2 + " is not equal to "+ parameterParser);
192    }

219    /**
220     * This Test method checks the DefaultParameterParser which carries two Sets inside it.
221     * The suggested problem was that pp.keySet() returns both Keys, but pp.getStrings("key")
222     * only checks for keys which are not Parts.
223     *
224     * @throws Exception generic exception
225     */
226    @Test
227    public void testAddPathInfo() throws Exception
228    {
229        assertEquals(0, parameterParser.keySet().size(), "keySet() is not empty!");
230
231        // Push this into the parser using DefaultParameterParser's add() method.
232        ((DefaultParameterParser) parameterParser).add("upload-field", mockTest);
233
234        assertEquals(1, parameterParser.keySet().size(), "Part not found in keySet()!");
235
236        Iterator<String> it = parameterParser.keySet().iterator();
237        assertTrue(it.hasNext());
238
239        String name = it.next();
240        assertEquals( "upload-field", name,"Wrong name found");
241
242        assertFalse(it.hasNext());
243
244        parameterParser.add("other-field", "foo");
245
246        assertEquals( 2, parameterParser.getKeys().length, "Wrong number of fields found ");
247
248        assertTrue(parameterParser.containsKey("upload-field"));
249        assertTrue(parameterParser.containsKey("other-field"));
250
251        // The following will actually cause a ClassCastException because getStrings() (and others) are not catering for Parts.
252        assertNull(parameterParser.getStrings("upload-field"), "The returned should be null because a Part is not a String");
253        assertFalse(parameterParser.containsKey("missing-field"));
254        
255        // The following will actually cause a ClassCastException because getPart() (and others) are not catering for Non-Parts, e.g String.
256        assertNull(parameterParser.getPart( "other-field" ), "The returned should be null because a String is not a Part");
257        Part uploadField = parameterParser.getPart( "upload-field" );
258        assertEquals("upload-field", uploadField.getName());
259
260    }
