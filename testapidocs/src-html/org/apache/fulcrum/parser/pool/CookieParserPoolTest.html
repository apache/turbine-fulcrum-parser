001package org.apache.fulcrum.parser.pool;

003
004import static org.junit.jupiter.api.Assertions.assertTrue;
005import static org.junit.jupiter.api.Assertions.fail;
006import static org.mockito.Mockito.mock;
007
008import javax.servlet.http.HttpServletRequest;
009import javax.servlet.http.HttpServletResponse;
010
011import org.apache.avalon.framework.component.ComponentException;
012import org.apache.commons.pool2.impl.GenericObjectPoolConfig;
013import org.apache.fulcrum.parser.DefaultCookieParser;
014import org.apache.fulcrum.parser.ParserService;
015import org.apache.fulcrum.testcontainer.BaseUnit5Test;
016import org.junit.jupiter.api.AfterEach;
017import org.junit.jupiter.api.BeforeEach;
018import org.junit.jupiter.api.Test;
019
020
021
022/**
023 * Test the CookieParserFactory and CookieParserPool classes.
024 *
025 * @author <a href="mailto:painter@apache.org">Jeffery Painter</a>
026 * @version $Id: CookieParserPoolTest.java 222043 2019-01-17 08:17:33Z painter $
027 */
028public class CookieParserPoolTest extends BaseUnit5Test 
029{
030        private DefaultCookieParser parser;
031    private ParserService parserService;
032    
033    /** 
034     * Use commons pool to manage value parsers 
035     */
036    private CookieParserPool cookieParserPool;
037
038    /**
039     * Performs any initialization that must happen before each test is run.
040     * @throws Exception if parser service not found
041     */
042    @BeforeEach
043    public void setUp() throws Exception
044    {
045        try
046        {
047            parserService = (ParserService)this.lookup(ParserService.ROLE);
048            
049
050                // Define the default configuration
051                GenericObjectPoolConfig config = new GenericObjectPoolConfig();
052                config.setMaxIdle(1);
053            config.setMaxTotal(1);
054
055            // init the pool
056            cookieParserPool 
057                = new CookieParserPool(new CookieParserFactory(), config);
058
059        }
060        catch (ComponentException e)
061        {
062            e.printStackTrace();
063            fail(e.getMessage());
064        }
065    }
066
067    /**
068     * Clean up after each test is run.
069     */
070    @AfterEach
071    public void tearDown()
072    {
073        // pool object already explicitely released by call to returnObject in test
074        // will throw java.lang.IllegalStateException, as pool is external to parserService
075        //parserService.putParser(parser);
076        this.release(parserService);
077    }
078    
079    /**
080     * @throws Exception generic exception
081     */
082    @Test
083    public void testFactoryMethods() throws Exception 
084    {
085        try
086        {
087                // borrow a new parser and assign it to the parser service
088                parser = cookieParserPool.borrowObject();
089                parser.setParserService(parserService);
090                
091            // Populate parser with mock servlet data
092            HttpServletRequest request = getMockRequest();
093            HttpServletResponse response = mock(HttpServletResponse.class);
094            parser.setData(request, response);
095                
096                // test setting cookies
097                parser.set("test1", "val1");
098
099                // clear the parser for reset
100                parser.clear();
101                assertTrue(parser.isValid());
102                
103                cookieParserPool.returnObject( parser );
104                
105        } catch ( Exception e )
106        {
107                e.printStackTrace();
108                fail(e.getMessage());
109        }
110    }
111
112}
