022import java.util.List;
023
024import javax.servlet.http.HttpServletRequest;
025import javax.servlet.http.Part;
026
027import org.apache.avalon.framework.service.ServiceException;
028import org.apache.fulcrum.parser.ValueParser.URLCaseFolding;
029
030/**
031 * ParserService defines the methods which are needed by the parser objects
032 * to get their necessities.
033 *
034 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
035 * @version $Id: ValueParser.java 535465 2007-05-05 06:58:06Z tv $
036 */
037public interface ParserService
038{
039    /** Avalon Identifier **/
040    String ROLE = ParserService.class.getName();
041
042    /** Default Encoding for Parameter Parser */
043    String PARAMETER_ENCODING_DEFAULT = "ISO-8859-1";
044
045    /** Key for the Parameter Parser Encoding */
046    String PARAMETER_ENCODING_KEY = "parameterEncoding";
047
048    /** Property for setting the URL folding value */
049    String URL_CASE_FOLDING_KEY = "urlCaseFolding";
050
051    /** Parse file upload items automatically */
052    String AUTOMATIC_KEY = "automaticUpload";
053    
054    /** fulcrum pool by default false */
055    boolean FULCRUM_POOL_DEFAULT = false;
056    
057    /** fulcrum pool activation parameter */
058    String FULCRUM_POOL_KEY = "fulcrumPool";
059    
060    /** commons pool2 parameters */
061    String POOL_KEY = "pool2";
062
063    /**
064     * <p> The default value of 'automaticUpload' property
065     * (<code>false</code>).  If set to <code>true</code>, parsing the
066     * multipart request will be performed automatically by {@link
067     * org.apache.fulcrum.parser.ParameterParser}.  Otherwise, an 
068     * org.apache.turbine.modules.Action may decide to parse the
069     * request by calling {@link #parseUpload(HttpServletRequest)
070     * parseRequest} manually.
071     */
072    boolean AUTOMATIC_DEFAULT = false;
073    
074    /**
075     * <p> The default value of 'maxTotal' property in 'pool'
076     * (<code>1024</code>). The default pool capacity.
077     */
078    int DEFAULT_POOL_CAPACITY = 1024;
079    
080    /**
081     * <p> The default value of 'maxIdle' property in 'pool'
082     * (<code>2</code>). The default maximum idle object.
083     */
084    int DEFAULT_MAX_IDLE = 2;
085
086    /**
087     * Get the parameter encoding that has been configured as default for
088     * the ParserService.
089     * 
090     * @return A String for the parameter encoding
091     */
092    String getParameterEncoding();
093
094    /**
095     * Trims the string data and applies the conversion specified in
096     * the property given by URL_CASE_FOLDING. It returns a new
097     * string so that it does not destroy the value data.
098     *
099     * @param value A String to be processed.
100     * @return A new String converted to lowercase and trimmed.
101     */
102    String convert(String value);
103
104    /**
105     * Convert a String value according to the url-case-folding property.
106     *
107     * @param value the String to convert
108     * @return a new String.
109     *
110     */
111    String convertAndTrim(String value);
112
113    /**
114     * A convert method, which trims the string data and applies the
115     * conversion specified in the parameter given. It returns a new
116     * string so that it does not destroy the value data.
117     *
118     * @param value A String to be processed.
119     * @param fold The parameter folding to be applied
120     * (see {@link ParserService})
121     * @return A new String converted to the correct case and trimmed.
122     */
123    String convertAndTrim(String value, URLCaseFolding fold);
124
125    /**
126     * Gets the folding value from the configuration
127     *
128     * @return The current Folding Value
129     */
130    URLCaseFolding getUrlFolding();
131
132    /**
133     * Gets the automaticUpload value from the configuration
134     *
135     * @return The current automaticUpload Value
136     */
137    boolean getAutomaticUpload();
138
139    /**
140     * Parse the given request for uploaded files
141     *
142     * @param request the HttpServletRequest object
143     * @return A list of {@link javax.servlet.http.Part}s
144     * @throws ServiceException if parsing fails
145     */
146    List<Part> parseUpload(HttpServletRequest request) throws ServiceException;
147
148    /**
149     * Get a {@link ValueParser} instance from the service. Use the
150     * default implementation.
151     *
152     * @param <P> The ValueParser we are using
153     * @param ppClass parameter parser class
154     * @return An object that implements ValueParser
155     * @throws InstantiationException if the instance could not be created
156     */
157    <P extends ValueParser> P getParser(Class<P> ppClass) throws InstantiationException;
158
159    /**
160     * Put the parser into service
161     *
162     * @param parser The value parser to be used 
163     */
164    void putParser(ValueParser parser);
165}
